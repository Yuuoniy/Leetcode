bool check(int post[],int postCount,int temp,int sum){

	if(temp==1){
		for(int i=1;i<=postCount;i++){
			if(post[i]==sum) return true;
		}
	}
	else
		for(int j=1;j<=postCount;j++){
			if(check(post,postCount,temp-1,sum-post[j]))
				return true;
		}
	return false;
}
