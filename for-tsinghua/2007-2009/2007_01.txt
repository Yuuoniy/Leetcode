#include <fstream>
#include <iostream>
using namespace std;
bool Zi(int i){

	for(int j=2;j<i+1;j++)
		if(i%j==0)return true;
	return false;
}
int main(){

	ifstream in("input.txt");
		if(!in){
			cout << "cannot open the file";
			return 1;
		}
	int n,i;
	in >> n;
	in.close();
	int count = 1;
	for(i=2;i<n/2+1;i++)
		if(Zi(i)){
			while(n%i==0){
				n = n/i;
				count++;
			}
		}
	ofstream out("output.txt");
	out << count;
	out.close();
	return 0;
}
