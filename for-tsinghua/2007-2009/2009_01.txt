#include "stdio.h"
#include "string.h"


int main(){
	
	int a0;
	int a1;
	scanf("%d",&a0);
	scanf("%d",&a1);
	int r,s;
	scanf("%d",&r);
	scanf("%d",&s);	

	int result;
	int k;

	scanf("%d",&k);
	if(k==0) {printf("%d",a0);return 0;}
	if(k==1) {printf("%d",a1);return 0;}
	while(k>1){
		k--;
		result = a1*r + a0*s;
		a0 = a1;
		a1 = result%10000;
	}
	printf("%d",a1);
	return 0;

}
